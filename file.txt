variables:
  ACCELQ_API_KEY:
    value: ""
    description: "Your ACCELQ API Key"
  ACCELQ_USER:
    value: ""
    description: "Your ACCELQ Email"
  ACCEL_DOMAIN:
    value: "navigate.accelq.io"
    description: "Your ACCELQ Domain"
  ACCEL_TENANT:
    value: "123"
    description: "Your ACCELQ Tenant ID"
  JIRA_DOMAIN:
    value: "jira.yourcompany.com"
    description: "Your Jira Server/DC URL (Do not include https://)"
  JIRA_PAT:
    value: ""
    description: "Your Jira Personal Access Token"
  ACCELQ_RUN_ID:
    value: ""
    description: "The ID of the ACCELQ run you want to sync"

stages:
  - sync_results

sync_to_xray:
  stage: sync_results
  image: alpine:latest
  before_script:
    # Ensures both curl and jq are available for the script
    - apk add --no-cache curl jq
  script:
    - echo "Fetching ACCELQ Run Data for Run ID $ACCELQ_RUN_ID..."
    # 1. Fetch ACCELQ Run Data
    - |
      RUN_DATA=$(curl -s -X GET "https://$ACCEL_DOMAIN/api/1.0/$ACCEL_TENANT/runs/$ACCELQ_RUN_ID" \
      -H "api_key: $ACCELQ_API_KEY" \
      -H "user_id: $ACCELQ_USER")

    # 2. Logic: Detect Target (Mobile/Browser) from the runName
    - |
      RUN_NAME=$(echo "$RUN_DATA" | jq -r '.runName')
      echo "Detected Run Name: $RUN_NAME"
      if [[ "$RUN_NAME" =~ [Mm]obile ]]; then
        TARGET="Mobile"
      else
        TARGET="Browser"
      fi
      echo "Targeting Jira tickets containing: $TARGET"

    # 3. Loop through Scenarios
    - |
      echo "$RUN_DATA" | jq -c '.testCases[]' | while read -r scenario; do
        STATUS=$(echo "$scenario" | jq -r '.testStatus')
        if [ "$STATUS" == "PASS" ]; then
          XRAY_STATUS="PASS"
        else
          XRAY_STATUS="FAIL"
        fi
        
        ALL_KEYS=$(echo "$scenario" | jq -r '.jiraIssues[]')
        TEST_KEYS_JSON=""
        EXECUTION_KEY=""

        for KEY in $ALL_KEYS; do
          # Use JIRA_PAT to check Jira Issue Type
          JIRA_DATA=$(curl -s -H "Authorization: Bearer $JIRA_PAT" \
            -X GET "https://$JIRA_DOMAIN/rest/api/2/issue/$KEY?fields=issuetype,summary")
          
          TYPE=$(echo "$JIRA_DATA" | jq -r '.fields.issuetype.name')
          SUMMARY=$(echo "$JIRA_DATA" | jq -r '.fields.summary')

          if [ "$TYPE" == "Test Execution" ] && echo "$SUMMARY" | grep -iq "$TARGET"; then
            EXECUTION_KEY=$KEY
          elif [ "$TYPE" == "Test" ]; then
            TEST_KEYS_JSON="${TEST_KEYS_JSON}{\"testKey\":\"$KEY\",\"status\":\"$XRAY_STATUS\"},"
          fi
        done

        # 4. Push to Xray Server API
        if [ ! -z "$EXECUTION_KEY" ] && [ ! -z "$TEST_KEYS_JSON" ]; then
          # Remove trailing comma from JSON array
          TEST_KEYS_JSON=$(echo "$TEST_KEYS_JSON" | sed 's/,$//')
          PAYLOAD="{\"testExecutionKey\":\"$EXECUTION_KEY\",\"tests\":[$TEST_KEYS_JSON]}"
          
          echo "Pushing results to Xray Server for Execution: $EXECUTION_KEY"
          curl -s -H "Authorization: Bearer $JIRA_PAT" \
               -H "Content-Type: application/json" \
               -X POST -d "$PAYLOAD" \
               "https://$JIRA_DOMAIN/rest/xray/2.0/api/import/execution"
        fi
      done

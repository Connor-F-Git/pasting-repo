variables:
  ACCELQ_API_KEY:
    value: ""
    description: "Your ACCELQ API Key"
  ACCELQ_USER:
    value: ""
    description: "Your ACCELQ Email"
  JIRA_USER_EMAIL:
    value: ""
    description: "Your Jira Email"
  JIRA_API_TOKEN:
    value: ""
    description: "Your Jira API Token"
  XRAY_ID:
    value: ""
    description: "Your Xray Client ID"
  XRAY_SECRET:
    value: ""
    description: "Your Xray Client Secret"
  ACCEL_DOMAIN:
    value: "navigate.accelq.io"
    description: "Your ACCELQ Domain"
  ACCEL_TENANT:
    value: "123"
    description: "Your ACCELQ Tenant ID"
  JIRA_DOMAIN:
    value: "company.atlassian.net"
    description: "Your Jira URL"

stages:
  - sync_results

sync_to_xray:
  stage: sync_results
  image: curlimages/curl:latest
  script:
    # 1. Fetch ACCELQ Run Data (ACCELQ_RUN_ID comes from the webhook payload or manual input)
    - |
      RUN_DATA=$(curl -s -X GET "https://$ACCEL_DOMAIN/api/1.0/$ACCEL_TENANT/runs/$ACCELQ_RUN_ID" \
      -H "api_key: $ACCELQ_API_KEY" \
      -H "user_id: $ACCELQ_USER")

    # 2. Logic: Detect if this is a Mobile or Browser run from the Run Name
    - |
      RUN_NAME=$(echo "$RUN_DATA" | jq -r '.runName')
      echo "Detected Run: $RUN_NAME"
      [[ "$RUN_NAME" =~ [Mm]obile ]] && TARGET="Mobile" || TARGET="Browser"
      echo "Only updating Jira tickets containing: $TARGET"

    # 3. Get Xray Auth Token
    - |
      XRAY_TOKEN=$(curl -s -H "Content-Type: application/json" \
      -X POST --data "{ \"client_id\": \"$XRAY_ID\", \"client_secret\": \"$XRAY_SECRET\" }" \
      https://xray.cloud.getxray.app/api/v2/authenticate | tr -d '"')

    # 4. Loop through Scenarios
    - |
      echo "$RUN_DATA" | jq -c '.testCases[]' | while read scenario; do
        STATUS=$(echo $scenario | jq -r '.testStatus')
        [ "$STATUS" == "PASS" ] && XRAY_STATUS="PASSED" || XRAY_STATUS="FAILED"
        
        ALL_KEYS=$(echo $scenario | jq -r '.jiraIssues[]')
        TEST_KEYS_JSON=""
        EXECUTION_KEY=""

        for KEY in $ALL_KEYS; do
          JIRA_DATA=$(curl -s -u "$JIRA_USER_EMAIL:$JIRA_API_TOKEN" \
            -X GET "https://$JIRA_DOMAIN/rest/api/2/issue/$KEY?fields=issuetype,summary")
          
          TYPE=$(echo "$JIRA_DATA" | jq -r '.fields.issuetype.name')
          SUMMARY=$(echo "$JIRA_DATA" | jq -r '.fields.summary')

          if [ "$TYPE" == "Test Execution" ] && echo "$SUMMARY" | grep -iq "$TARGET"; then
            EXECUTION_KEY=$KEY
          elif [ "$TYPE" == "Test" ]; then
            TEST_KEYS_JSON="${TEST_KEYS_JSON}{\"testKey\":\"$KEY\",\"status\":\"$XRAY_STATUS\"},"
          fi
        done

        # 5. Push to Xray
        if [ ! -z "$EXECUTION_KEY" ] && [ ! -z "$TEST_KEYS_JSON" ]; then
          TEST_KEYS_JSON=$(echo $TEST_KEYS_JSON | sed 's/,$//')
          PAYLOAD="{\"testExecutionKey\":\"$EXECUTION_KEY\",\"tests\":[$TEST_KEYS_JSON]}"
          
          curl -s -H "Authorization: Bearer $XRAY_TOKEN" \
               -H "Content-Type: application/json" \
               -X POST --data "$PAYLOAD" \
               "https://xray.cloud.getxray.app/api/v2/import/execution"
          echo "Updated $EXECUTION_KEY for tests associated with this scenario."
        fi
      done
